// ============================================================
//  Умный термостат на Arduino UNO с датчиком DHT11 и LCD 16x2
//  Светодиод (LED) имитирует включение нагревателя
// ============================================================

// ------ ПОДКЛЮЧЕНИЕ БИБЛИОТЕК (поставьте их через менеджер библиотек) ------
#include <LiquidCrystal_I2C.h>   // Библиотека для работы с LCD-дисплеем по I2C
#include <DHT.h>                 // Библиотека для работы с датчиком DHT11/DHT22

// ------ НАСТРОЙКА ПИНОВ (куда физически подключены компоненты) ------
#define DHTPIN 2          // Датчик DHT11 подключён к цифровому пину D2
#define DHTTYPE DHT11     // Указываем тип датчика (DHT11). Если DHT22 — измените
#define LEDPIN 3          // Светодиод (нагреватель) подключён к пину D3

// ------ СОЗДАНИЕ ОБЪЕКТОВ (экземпляров классов) ------
DHT dht(DHTPIN, DHTTYPE);                    // Создаём объект dht для работы с датчиком
LiquidCrystal_I2C lcd(0x27, 16, 2);         // Создаём объект lcd: 
                                             // 0x27 — I2C-адрес дисплея (может быть 0x3F)
                                             // 16 — символов в строке, 2 — строки

// ------ ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ (настройки, которые можно менять) ------
float tempOn = 25.0;      // Температура включения нагрева (если t > 25°C — греем)

// ------ ФУНКЦИЯ setup() — выполняется 1 раз при включении или сбросе ------
void setup() {
  dht.begin();            // Запускаем датчик DHT (инициализация)
  
  lcd.init();             // Инициализация дисплея (подготовка к работе)
  lcd.backlight();        // Включаем подсветку экрана (чтобы видеть текст)
  
  pinMode(LEDPIN, OUTPUT);          // Настраиваем пин светодиода как ВЫХОД
  digitalWrite(LEDPIN, LOW);        // Гарантированно выключаем светодиод при старте

  // Приветственное сообщение при загрузке
  lcd.print("Temp System");         // Выводим текст на экран
  delay(2000);                      // Ждём 2 секунды (чтобы пользователь увидел)
  lcd.clear();                      // Очищаем экран перед основным циклом
}

// ------ ФУНКЦИЯ loop() — выполняется бесконечно после setup() ------
void loop() {
  delay(2000);                      // Ждём 2 секунды между опросами (чтобы не спамить)

  float t = dht.readTemperature();  // Читаем температуру с датчика (в градусах Цельсия)

  // ------ ПРОВЕРКА ОШИБОК ДАТЧИКА ------
  if (isnan(t)) {                   // Если датчик вернул "не число" (ошибка/обрыв)
    lcd.setCursor(0, 0);            // Ставим курсор в начало первой строки
    lcd.print("Sensor Error!");     // Выводим сообщение об ошибке
    return;                         // Выходим из loop() до следующего опроса
  }

  // ------ ВЫВОД ТЕКУЩЕЙ ТЕМПЕРАТУРЫ НА ЭКРАН ------
  lcd.setCursor(0, 0);              // Курсор в начало первой строки
  lcd.print("Temp: ");              // Выводим текст "Temp: "
  lcd.print(t);                     // Выводим значение температуры (например, 23.5)
  lcd.print(" C  ");                // Выводим " C" и два пробела (чтобы стереть лишние символы)

  // ------ УПРАВЛЕНИЕ НАГРЕВАТЕЛЕМ (сравнение с порогом) ------
  if (t > tempOn) {                 // Если температура ВЫШЕ порога (25°C)
    digitalWrite(LEDPIN, HIGH);     // Включаем светодиод (имитация нагрева)
    lcd.setCursor(0, 1);            // Курсор во вторую строку
    lcd.print("HEATING ON  ");      // Выводим статус "НАГРЕВ ВКЛ" + пробелы для очистки
  } else {                          // Иначе (температура НИЖЕ или РАВНА порогу)
    digitalWrite(LEDPIN, LOW);      // Выключаем светодиод
    lcd.setCursor(0, 1);            // Курсор во вторую строку
    lcd.print("HEATING OFF ");      // Выводим статус "НАГРЕВ ВЫКЛ"
  }
} // Конец loop() — далее цикл повторяется заново
